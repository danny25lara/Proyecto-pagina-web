function calcularBono() {
  var hijos = parseInt(document.getElementById('hijos').value);
  var trabajo = document.getElementById('trabajo').value;
  var bono = 0;

  if (isNaN(hijos) || hijos < 0 || trabajo === "") {
    alert("Completa todos los campos correctamente.");
    return;
  }

  if (trabajo === "si") {
    bono = hijos * 50;
  } else {
    bono = hijos * 30;
  }

  document.getElementById('resultado').textContent =
    "Tu bono es: $" + bono;
  document.getElementById('resultado').style.opacity = 1;
}
