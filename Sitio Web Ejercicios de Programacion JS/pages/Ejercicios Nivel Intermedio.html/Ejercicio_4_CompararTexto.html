🔤 Comparar longitud de dos textos

Escribe dos textos y el sistema te dirá cuál tiene más letras, o si son iguales. 

function compararTextos() {
const t1 = document.getElementById("texto1").value.trim();
const t2 = document.getElementById("texto2").value.trim();

let mensaje = "";

if (t1.length > t2.length) {
mensaje = "El primer texto tiene más letras.";
} else if (t2.length > t1.length) {
mensaje = "El segundo texto tiene más letras.";
} else {
mensaje = "Ambos textos tienen la misma cantidad de letras.";
}
document.getElementById("resultado").textContent = mensaje;
document.getElementById("resultado").classList.add("mostrar");
}
