✂️ Separar palabras de una oración

Escribe una oración y te mostraremos cada palabra por separado. 

function dividirPalabras() {
  const oracion = document.getElementById("oracion").value.trim();
  const palabras = oracion.split(" ");
  const resultado = palabras.join("\\n");

  document.getElementById("resultado").textContent = resultado;
  document.getElementById("resultado").classList.add("mostrar");
}
