Escribe tu nombre completo y te diremos tus iniciales. 

function mostrarIniciales() {
const nombre = document.getElementById("nombre").value.trim();
const partes = nombre.split(/\s+/);
let iniciales = "";
for (let i = 0; i < partes.length; i++) {
if (partes[i].length > 0) {
    iniciales += partes[i][0].toUpperCase() + ".";
}
}
document.getElementById("resultado").textContent = "Iniciales: " + iniciales;
document.getElementById("resultado").classList.add("mostrar");
}
