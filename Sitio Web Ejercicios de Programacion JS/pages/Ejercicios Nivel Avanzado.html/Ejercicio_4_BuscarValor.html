Ingresa los elementos del arreglo y el valor a buscar.
Ejemplo del arreglo: manzana, pera, mango

function buscarValor() {
  const arregloInput = document.getElementById('inputArreglo').value.trim();
  const valorBuscado = document.getElementById('inputValor').value.trim().toLowerCase();

  const arreglo = arregloInput.split(',').map(e => e.trim().toLowerCase());
  const existe = arreglo.includes(valorBuscado);

  const mensaje = existe
    ? '‚úÖ El valor "' + valorBuscado + '" S√ç existe en el arreglo.'
    : '‚ùå El valor "' + valorBuscado + '" NO se encuentra en el arreglo.';

  document.getElementById('resultado').textContent = mensaje;
  document.getElementById('resultado').classList.add('mostrar');
}
